**JavaScript** (in {% block scripts %}):
Three functions:

```javascript
async function testStep(service, resultId, btn) {
    const el = document.getElementById(resultId);
    el.textContent = 'Verificando...';
    btn.disabled = true;
    try {
        const r = await fetch('/admin/ui/onboarding/test-step', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({service: service})
        });
        const data = await r.json();
        el.textContent = data.ok ? '\u2705 ' + data.detail : '\u274c ' + data.detail;
        el.style.color = data.ok ? 'var(--success)' : 'var(--danger)';
    } catch (e) {
        el.textContent = '\u274c Error de red';
        el.style.color = 'var(--danger)';
    } finally {
        btn.disabled = false;
    }
}

async function sendTestMessage(btn) {
    const phone = document.getElementById('test-phone').value.trim();
    const el = document.getElementById('send-result');
    if (!phone) { el.textContent = 'Ingresa el numero primero'; el.style.color='var(--danger)'; return; }
    el.textContent = 'Enviando...';
    btn.disabled = true;
    try {
        const r = await fetch('/admin/ui/onboarding/send-test', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({phone: phone})
        });
        const data = await r.json();
        el.textContent = data.ok ? '\u2705 ' + data.detail : '\u274c ' + data.detail;
        el.style.color = data.ok ? 'var(--success)' : 'var(--danger)';
    } catch (e) {
        el.textContent = '\u274c Error de red';
        el.style.color = 'var(--danger)';
    } finally {
        btn.disabled = false;
    }
}

function confirmReceived() {
    // Just reload — completion is derived from DB state (no server call needed per D-05)
    location.reload();
}
```
