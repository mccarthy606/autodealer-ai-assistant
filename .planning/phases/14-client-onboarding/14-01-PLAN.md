```python
async def onboarding_page(request: Request, db: AsyncSession = Depends(get_db)):
    did = await auth_check(request)
    if not isinstance(did, int): return did
    stmt = select(Dealership).where(Dealership.id == did)
    r = await db.execute(stmt)
    dealer = r.scalar_one_or_none()
    # Step completion per D-05 (computed fresh each GET, no wizard_state column):
    step1_done = bool(dealer and dealer.whatsapp_access_token and dealer.whatsapp_phone_number_id)
    step2_done = bool(dealer and dealer.ml_access_token and dealer.ml_user_id)
    step3_done = bool(dealer and dealer.ml_last_sync_at)
    step4_done = step1_done and step2_done and step3_done  # always re-doable per D-05
    # Active step = first incomplete step (1-4), or 4 if all done:
    if not step1_done: active_step = 1
    elif not step2_done: active_step = 2
    elif not step3_done: active_step = 3
    else: active_step = 4
    saved = request.query_params.get("saved")
    return templates.TemplateResponse("admin/onboarding.html", {
        "request": request, "dealer": dealer,
        "step1_done": step1_done, "step2_done": step2_done,
        "step3_done": step3_done, "step4_done": step4_done,
        "active_step": active_step, "saved": saved,
    })
```

**Route 2 — POST /admin/ui/onboarding/save**:
```python
async def onboarding_save(request: Request, db: AsyncSession = Depends(get_db)):
    did = await auth_check(request)
    if not isinstance(did, int): return did
    form = await request.form()
    step = int(form.get("step", 0))
    stmt = select(Dealership).where(Dealership.id == did)
    r = await db.execute(stmt)
    dealer = r.scalar_one_or_none()
    if dealer:
        if step == 1:
            if form.get("whatsapp_phone_number_id"):
                dealer.whatsapp_phone_number_id = form["whatsapp_phone_number_id"].strip()
            if form.get("whatsapp_access_token"):
                dealer.whatsapp_access_token = form["whatsapp_access_token"].strip()
            if form.get("whatsapp_verify_token"):
                dealer.whatsapp_verify_token = form["whatsapp_verify_token"].strip()
            if form.get("whatsapp_webhook_secret"):
                dealer.whatsapp_webhook_secret = form["whatsapp_webhook_secret"].strip()
        elif step == 2:
            if form.get("ml_access_token"):
                dealer.ml_access_token = form["ml_access_token"].strip()
            if form.get("ml_refresh_token"):
                dealer.ml_refresh_token = form["ml_refresh_token"].strip()
            if form.get("ml_app_id"):
                dealer.ml_app_id = form["ml_app_id"].strip()
            if form.get("ml_client_secret"):
                dealer.ml_client_secret = form["ml_client_secret"].strip()
            if form.get("ml_user_id"):
                dealer.ml_user_id = form["ml_user_id"].strip()
    await db.commit()
    return RedirectResponse(url=f"/admin/ui/onboarding?saved={step}", status_code=302)
```

```python
async def onboarding_send_test(request: Request, db: AsyncSession = Depends(get_db)):
    did = await auth_check(request)
    if not isinstance(did, int):
        return JSONResponse({"ok": False, "detail": "No autenticado"})
    try:
        body = await request.json()
    except Exception:
        return JSONResponse({"ok": False, "detail": "Solicitud invalida"})
    phone = (body.get("phone") or "").strip()
    if not phone:
        return JSONResponse({"ok": False, "detail": "Numero requerido"})
    stmt = select(Dealership).where(Dealership.id == did)
    r = await db.execute(stmt)
    dealer = r.scalar_one_or_none()
    if not dealer or not dealer.whatsapp_access_token or not dealer.whatsapp_phone_number_id:
        return JSONResponse({"ok": False, "detail": "WhatsApp no configurado -- completa el Paso 1 primero"})
    adapter = WhatsAppCloudAdapter(
        phone_number_id=dealer.whatsapp_phone_number_id,
        token=dealer.whatsapp_access_token,
    )
    result = await adapter.send_text(
        phone,
        "Hola, este es un mensaje de prueba de tu bot AutoDealer. ¡Todo funciona correctamente! 🚗"
    )
    if result.get("error"):
        return JSONResponse({"ok": False, "detail": f"Error al enviar: {result['error']}"})
    return JSONResponse({"ok": True, "detail": "Mensaje enviado. ¡Revisa tu WhatsApp!"})
```
